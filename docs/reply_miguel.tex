\subsubsection*{Training}

\begin{itemize}[nosep]
  \item Adam optimizer, constant learning rate $10^{-4}$.
  \item Batch size 16, sequence length 128.
  \item 150 epochs, no early stopping --- we save the best validation loss checkpoint.
  \item 70/15/15 temporal split (train/val/test).
  \item Trained on CPU.
\end{itemize}

Let $d$ denote the number of assets ($d = 3$).
Each extreme event $i$ is a mark
$(\boldsymbol{W}_i, R_i, \Delta T_i)$
where $\boldsymbol{W}_i \in \mathbb{S}^{d-1}$ is the angular direction,
$R_i > 0$ the radial magnitude,
and $\Delta T_i = T_i - T_{i-1} > 0$ the inter-event time.
The direction-dependent threshold $u_\tau(\boldsymbol{W})$ is
learned via quantile regression at level $\tau = 0.95$;
an observation is extreme when $R > u_\tau(\boldsymbol{W})$.

Each event is tokenised as
$\boldsymbol{x}_i = (\boldsymbol{W}_i,\; \log R_i,\; \log \Delta T_i)
  \in \mathbb{R}^{5}$.
A causal Transformer encoder (4 layers,
$d_{\text{model}} = 128$, 4 attention heads per layer,
sinusoidal positional encoding) maps the history
$(\boldsymbol{x}_1, \ldots, \boldsymbol{x}_i)$ to a hidden state
$h_i = f_\theta(\boldsymbol{x}_1, \ldots, \boldsymbol{x}_i)
  \in \mathbb{R}^{128}$.
Three prediction heads (MLPs) then produce, from $h_i$,
the parameters of the next-event densities described below.

\subsubsection*{Loss}

The model minimises the negative joint log-likelihood:
%
\[
  \mathcal{L}
  = -\frac{1}{N}\sum_{i=1}^{N}
    \bigl\{
      \log p(\boldsymbol{W}_i \mid h_i)
    + \log p(R_i \mid h_i, \boldsymbol{W}_i)
    + \log p(\Delta T_i \mid h_i)
    \bigr\}
\]
%
where each term corresponds to one prediction head:
%
\begin{itemize}[nosep]
  \item $p(\boldsymbol{W} \mid h)$ is a mixture of $K{=}6$ von Mises--Fisher
        densities on $\mathbb{S}^{d-1}$ with weights $\pi_k$,
        mean directions $\boldsymbol{\mu}_k$,
        and concentrations $\kappa_k$.
  \item $p(R \mid h, \boldsymbol{W})$ is a Gamma density with
        shape $a=d$ and rate $\beta = g(\boldsymbol{W}; h)$, truncated below the
        directional threshold $u_\tau(\boldsymbol{W})$.
  \item $p(\Delta T \mid h)$ is an exponential density with Hawkes intensity
        $\lambda$.
\end{itemize}

\subsubsection*{Generation}

Given a trigger event $(\boldsymbol{W}_0, R_0)$ with
$R_0 > u_\tau(\boldsymbol{W}_0)$ and a time horizon $H > 0$,
the model generates a cascade autoregressively.

\begin{algorithm}[H]
\caption{-}
\begin{algorithmic}[1]
\Require trigger $(\boldsymbol{W}_0, R_0)$,\; $H > 0$,\; trained encoder $f_\theta$
\State $T_0 \gets 0$,\; $\mathcal{H} \gets \{(\boldsymbol{W}_0, R_0, T_0)\}$,\; $i \gets 0$
\While{$T_i < H$}
  \State $h_i \gets f_\theta(\boldsymbol{x}_0, \ldots, \boldsymbol{x}_i)$
  \State $\boldsymbol{W}_{i+1} \sim p(\boldsymbol{W} \mid h_i)$
  \State $R_{i+1} \sim p(R \mid h_i,\, \boldsymbol{W}_{i+1})$
  \State $\Delta T_{i+1} \sim p(\Delta T \mid h_i)$
  \State $T_{i+1} \gets T_i + \Delta T_{i+1}$
  \State $\mathcal{H} \gets \mathcal{H} \cup
         \{(\boldsymbol{W}_{i+1}, R_{i+1}, \Delta T_{i+1})\}$
  \State $i \gets i + 1$
\EndWhile
\State \Return $\mathcal{H}$
\end{algorithmic}
\end{algorithm}

\subsubsection*{Transition matrix}

We report the \emph{dominant-asset} transition matrix at the
\textbf{95th percentile} threshold:
%
\[
  P_{ij} = \mathbb{P}(d_{t+1} = j \mid d_t = i),
\]
%
where $d_t = \arg\max_j |W_t^{(j)}|$ is the dominant asset at extreme
event~$t$ ($W_t^{(j)}$ denotes the $j$-th component of $\boldsymbol{W}_t$).
This is estimated by counting consecutive pairs $(d_t, d_{t+1})$ over the
sequence of extreme events.
